const createPerson = (name, age, gender) => ({
    name,
    age,
    gender
});


// TESTS:
console.log(createPerson("Alba", 5, "female"));
// { name: 'Alba', age: 5, gender: 'female' }

console.log(createPerson("Zodiac Hasbro", 56, "male"))
// { name: 'Zodiac Hasbro', age: 56, gender: 'male' }
